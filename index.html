Changes the option of a select element

Explanation: When the user changes the option of a select element

Javascript Vanilla

Code 
    function select(selectId, optionValToSelect){
        var selectElement = document.getElementById(selectId);
        var selectOptions = selectElement.options;
        for (var opt, j = 0; opt = selectOptions[j]; j++) {
            if (opt.value == optionValToSelect) {
                selectElement.selectedIndex = j;
                break;
            }
        }
    }

Jquery

Code 
    $( ".target" ).change(function() {
      alert( "Handler for .change() called." );
    });

    $( "#other" ).click(function() {
      $( ".target" ).change();
    });

Fade in

          
Explanation: Fade in an HTML element using jQuery

          

            
Javascript Vanilla

            

              Code
              
                
    function unfade(element) {
      var op = 0.1;  // initial opacity
      element.style.display = 'block';
      var timer = setInterval(function () {
          if (op >= 1){
              clearInterval(timer);
          }
          element.style.opacity = op;
          element.style.filter = 'alpha(opacity=' + op * 100 + ")";
          op += op * 0.1;
      }, 10);
    }

Fade out

          
Explanation: Fade out an HTML element using jQuery

          

            
Javascript Vanilla

            

              Code
              
                
    function fade(element) {
      var op = 1;  // initial opacity
      var timer = setInterval(function () {
          if (op <= 0.1){
              clearInterval(timer);
              element.style.display = 'none';
          }
          element.style.opacity = op;
          element.style.filter = 'alpha(opacity=' + op * 100 + ")";
          op -= op * 0.1;
      }, 50);
    }
                

              
            

          

          

            
Jquery

            

              Code
              
                
    $(".btn1").click(function(){
      $("p").fadeOut();
    });
